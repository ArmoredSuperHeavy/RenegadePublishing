% Roman Chapter Style for memoir
\renewcommand{\thechapter}{\MakeTextUppercase{\roman{chapter}}}  % This changes the chapter numbers to Roman numerals everywhere (contents, chapter headings, page headers, etc)

\makechapterstyle{Roman}{
  \setlength{\beforechapskip}{1\onelineskip}      % beware if changing \beforechapskip - the actual whitespace is \onelineskip + \beforechapskip
  \setlength{\midchapskip}{1\onelineskip}
  \setlength{\afterchapskip}{1\onelineskip}
  \renewcommand*{\chapnamefont}{\centering}       % add \scshape to this if your font supports small caps 
  \renewcommand*{\chaptitlefont}{\chapnamefont }  % can also use \MakeTextUppercase for all caps
  \renewcommand*{\chapnumfont}{\chapnamefont }
  \renewcommand*{\printchaptername}{\chapnumfont Chapter} 
  \renewcommand*{\chapternamenum}{ }                          % the space between "Chapter" and "X"
  \renewcommand*{\printchapternum}{\chapnumfont \thechapter}
  \renewcommand*{\printchaptertitle}{\chaptitlefont}
}
\chapterstyle{Roman}
